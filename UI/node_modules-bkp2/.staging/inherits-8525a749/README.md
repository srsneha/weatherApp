A dead simple way to do inheritance in JS.

    var inherits = require("inherits")

    function Animal () {
      this.alive = true
    }
    Animal.prototype.say = function (what) {
      console.log(what)
    }

    inherits(Dog, Animal)
    function Dog () {
      Dog.super.apply(this)
    }
    Dog.prototype.sniff = function () {
      this.say("sniff sniff")
    }
    Dog.prototype.bark = function () {
      this.say("woof woof")
